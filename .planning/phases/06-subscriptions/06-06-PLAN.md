1. **Create `apps/web/src/lib/api/admin-subscriptions.ts`** following the pattern from `lib/api/admin-payments.ts`:

   ```typescript
   import api from './client';

   export async function adminListSubscriptions(params?: {
     page?: number;
     limit?: number;
     status?: string;
     planType?: string;
   }) {
     const { data } = await api.get('/subscriptions/admin/list', { params });
     return data; // { success, data, pagination }
   }

   export async function adminActivateSubscription(input: {
     userId: string;
     planType: string;
     durationDays?: number;
   }) {
     const { data } = await api.post('/subscriptions/admin/activate', input);
     return data.data;
   }

   export async function adminListPlans() {
     const { data } = await api.get('/subscriptions/admin/plans');
     return data.data; // array of all plan configs
   }

   export async function adminCreatePlan(input: {
     planType: string;
     name: string;
     price: number;
     billingInterval: string;
     collectionLimit: number;
     commissionRate: number;
     trialDays?: number;
     isActive?: boolean;
     stripePriceId?: string;
   }) {
     const { data } = await api.post('/subscriptions/admin/plans', input);
     return data.data;
   }

   export async function adminUpdatePlan(id: string, input: Record<string, unknown>) {
     const { data } = await api.put(`/subscriptions/admin/plans/${id}`, input);
     return data.data;
   }
   ```
