   ```typescript
   import api from './client';

   export async function getPlans() {
     const { data } = await api.get('/subscriptions/plans');
     return data.data; // array of PlanConfig
   }

   export async function getSubscriptionStatus() {
     const { data } = await api.get('/subscriptions/status');
     return data.data; // subscription status object
   }

   export async function createCheckout(planConfigId: string) {
     const { data } = await api.post('/subscriptions/checkout', { planConfigId });
     return data.data; // { url, sessionId }
   }

   export async function createPortalSession() {
     const { data } = await api.post('/subscriptions/portal');
     return data.data; // { url }
   }

   export async function cancelSubscription() {
     const { data } = await api.post('/subscriptions/cancel');
     return data.data; // updated subscription
   }
   ```
